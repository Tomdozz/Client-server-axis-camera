Appendix 1: Analytics libraries

RAster Processing Primitives library (RAPP)

RAPP is a library of low-level image processing operations, targeting mainly video analytics and computer vision applications. The library provides efficient algorithms for common geometrical transformations, pixelwise operations, etc. RAPP is released under Lesser GNU General Public License v3 and entirely open-source. The complete documentation is available here: RAPP Developer's Manual and here: RAPP User's Manual.

Example of RAPP use

#include <rapp/rapp.h>
/* main comes here */
void
modify_image (media_frame *frame)
{
int height;
int width;
int stride;
unsigned char *data;
unsigned char *rapp_buffer = NULL;
height = capture_frame_height(frame);
width = capture_frame_width(frame);
stride = capture_frame_stride(frame);
data = capture_frame_data(frame);
rapp_buffer = (unsigned char*) rapp_malloc (width * height, 0);
/* Rotate an 8-bit image 90 degrees in counter-clockwise direction. */
rapp_rotate_ccw_u8(rapp_buffer, height, data, stride, width, height);
/* write the image to output... */
}

Fixmath library

Fixmath is a professional library of fixed-point math operations and functions under Lesser GNU General Public License v3. Experimental results show that the fixed-point multiply and transcendental functions are more than two orders of magnitude faster than their floating-point equivalents, while division and the algebraic functions are 10-50 times faster. The complete documentation is available here: Fixmath User's Manual

Example of filter implementation with Fixmath

#include <fixmath.h>
void *filter_fixmath(unsigned char *data, int width, int height)
{
int i;
int j;
fixed_t multiplier = fx_ftox((1.0f / 16.0f), 10);
unsigned char *filtered = malloc(width * height);
for (i = 0; i < height; i++) {
for (j = 0; j < width; j++) {
int pos = i * width + j;
fixed_t sum = 0;
if (i > 0 && j > 0 && i < (height - 1) && j < (width - 1)) {
fixed_t tmp;
/****************************************************
j-1 j j+1
i-1 1 | 2 | 1
---|---|--
i 2 | 4 | 2 filter values
---|---|--
i+1 1 | 2 | 1
*/
tmp = fx_itox(data[(i - 1) * width + (j - 1)], 10);
sum += fx_mulx(multiplier, tmp, 10);
tmp = fx_itox(2 * data[(i - 1) * width + (j)], 10);
sum += fx_mulx(multiplier, tmp, 10);
tmp = fx_itox(data[(i - 1) * width + (j + 1)], 10);
sum += fx_mulx(multiplier, tmp, 10);
tmp = fx_itox(2 * data[(i) * width + (j - 1)], 10);
sum += fx_mulx(multiplier, tmp, 10);
tmp = fx_itox(4 * data[(i) * width + (j)], 10);
sum += fx_mulx(multiplier, tmp, 10);
tmp = fx_itox(2 * data[(i) * width + (j + 1)], 10);
sum += fx_mulx(multiplier, tmp, 10);
tmp = fx_itox(data[(i + 1) * width + (j - 1)], 10);
sum += fx_mulx(multiplier, tmp, 10);
tmp = fx_itox(2 * data[(i + 1) * width + (j)], 10);
sum += fx_mulx(multiplier, tmp, 10);
tmp = fx_itox(data[(i + 1) * width + (j + 1)], 10);
sum += fx_mulx(multiplier, tmp, 10);
filtered[pos] = fx_xtoi(sum, 10);
} else {
filtered[pos] = data[pos];
}
}
}
return filtered;
}
